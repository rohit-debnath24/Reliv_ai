pythonimport hashlib

def generate_access_code(phone_number):
    """
    Generate deterministic 4-digit code from phone number
    Same phone = Same code (always)
    """
    
    # Hash the phone number
    hash_object = hashlib.md5(phone_number.encode())
    hash_hex = hash_object.hexdigest()
    
    # Take first 8 characters and convert to int
    hash_int = int(hash_hex[:8], 16)
    
    # Modulo to get 4-digit number (1000-9999)
    code = (hash_int % 9000) + 1000
    
    return str(code)

pythonclass KioskAntifraudSystem:
    
    def __init__(self):
        self.fraud_checks = [
            self.check_phone_duplicate,
            self.check_kiosk_limit,
            self.check_timing_pattern,
            self.check_otp_attempts,
            self.check_email_pattern,
            self.check_behavioral_signals
        ]
    
    def check_phone_duplicate(self, phone):
        """
        PRIMARY CHECK: Has this phone been used before?
        """
        user = db.get_user_by_phone(phone)
        
        if user and user.has_used_free_trial:
            return {
                "blocked": False,  # Don't block, just charge
                "action": "require_payment",
                "message": f"Welcome back! Last visit: {user.last_visit}",
                "price": 19
            }
        return {"blocked": False}
    
    def check_kiosk_limit(self, kiosk_id):
        """
        Limit free trials per kiosk per day
        """
        today_count = db.count_free_trials_today(kiosk_id)
        
        if today_count >= 50:
            return {
                "blocked": True,
                "message": "Daily limit reached. Try tomorrow or pay ₹9 to continue.",
                "bypass_price": 9
            }
        return {"blocked": False}
    
    def check_timing_pattern(self, phone, kiosk_id):
        """
        Detect if someone is creating multiple accounts rapidly
        """
        recent_attempts = db.get_attempts_last_hour(kiosk_id)
        
        # If more than 3 OTP requests from same kiosk in 1 hour
        if len(recent_attempts) >= 3:
            # Check if different phone numbers
            unique_phones = set([a.phone for a in recent_attempts])
            
            if len(unique_phones) >= 3:
                return {
                    "blocked": True,
                    "suspicious": True,
                    "message": "Too many signups from this kiosk. Please wait 30 minutes.",
                    "cooldown": 30  # minutes
                }
        
        return {"blocked": False}
    
    def check_otp_attempts(self, phone):
        """
        Prevent OTP spam (costs money to send)
        """
        otp_history = db.get_otp_history(phone, hours=24)
        
        if len(otp_history) >= 5:
            return {
                "blocked": True,
                "message": "Too many OTP requests. Try again tomorrow.",
                "reason": "otp_spam_protection"
            }
        
        return {"blocked": False}
    
    def check_email_pattern(self, email):
        """
        Detect disposable/fake emails
        """
        disposable_domains = [
            "tempmail.com", "10minutemail.com", "guerrillamail.com",
            "mailinator.com", "throwaway.email"
        ]
        
        if any(domain in email for domain in disposable_domains):
            return {
                "blocked": True,
                "message": "Please use a permanent email address.",
                "suggestion": "Use Gmail, Yahoo, or Outlook"
            }
        
        # Check for sequential pattern (test1@, test2@, test3@)
        import re
        if re.search(r'(test|fake|dummy)\d+', email.lower()):
            return {
                "blocked": True,
                "suspicious": True,
                "message": "Please use your real email address."
            }
        
        return {"blocked": False}
    
    def check_behavioral_signals(self, session_data):
        """
        Detect bot-like behavior
        """
        signals = {
            "form_completion_time": session_data.get("completion_time"),
            "mouse_movement": session_data.get("mouse_events"),
            "keyboard_pattern": session_data.get("typing_speed")
        }
        
        # If form filled in less than 20 seconds (impossible for human)
        if signals["form_completion_time"] < 20:
            return {
                "blocked": True,
                "message": "Please take your time to fill the form accurately.",
                "reason": "too_fast"
            }
        
        # If zero mouse movement (bot)

            }

                "show_duration": 10  # seconds
            }
        
        return {"public_notice": False}

pythondef validate_referral(referral_code, referee_phone, kiosk_id):
    """
    Validate referral with strict anti-fraud checks
    """
    
    # Parse code (format: NAME + 4-digit code)
    # Example: FAIZAN9876
    referrer_code = referral_code[-4:]  # Last 4 digits
    
    # Find referrer
    referrer = db.get_user_by_code(referrer_code)
    
    if not referrer:
        return {
            "valid": False,
            "error": "Invalid referral code"
        }
    
    # CHECK 1: Can't refer yourself
    if referrer.phone == referee_phone:
        return {
            "valid": False,
            "error": "Nice try! Can't refer yourself 😉"
        }
    
    # CHECK 2: Referrer must be active user
    if referrer.activity_score < 3:
        return {
            "valid": False,
            "error": "Your friend hasn't used FitBot yet. Ask them to try it first!"
        }
    
    # CHECK 3: Referee must be NEW
    if db.user_exists(referee_phone):
        return {
            "valid": False,
            "error": "You already have an account! Login instead."
        }
    
    # CHECK 4: Same kiosk (prevents online fraud)
    if referrer.last_kiosk != kiosk_id:
        return {
            "valid": False,
            "error": "You must visit the same kiosk as your friend.",
            "friend_kiosk": referrer.last_kiosk_location
        }
    
    # CHECK 5: Referral limit (max 5/day to prevent spam)
    today_referrals = db.count_referrals_today(referrer.phone)
    if today_referrals >= 5:
        return {
            "valid": False,
            "error": "Your friend has reached daily referral limit. Try tomorrow!"
        }
    
    # CHECK 6: Timing check (can't refer 10 people in 10 minutes)
    recent_referrals = db.get_referrals_last_hour(referrer.phone)
    if len(recent_referrals) >= 3:
        return {
            "valid": False,
            "error": "Too fast! Wait 1 hour between referrals."
        }
    
    # ALL CHECKS PASSED
    return {
        "valid": True,
        "referrer": referrer.name,
        "welcome_message": f"Welcome! {referrer.name} sent you here 👋"
    }

def generate_photo_upload_qr(user_code):

    session_id = str(uuid.uuid4())
    
    # Mobile upload URL
    upload_url = f"https://fitbot.in/upload/{user_code}/{session_id}"
    
    # Generate QR code
    qr = qrcode.QRCode(version=1, box_size=10, border=4)
    qr.add_data(upload_url)
    qr.make(fit=True)
    
    qr_image = qr.make_image(fill_color="black", back_color="white")
    
    return {
        "qr_image": qr_image,

        "expires_in": 300  # 5 minutes
